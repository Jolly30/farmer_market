# Clone via git (using terminal)
    git clone < https://github.com/Jolly30/farmer_market.git>
    cd farmer_market

# Create a Python virtual environment(vscode terminal)
    Mac/Linux
    python3 -m venv myenv
    Windows
    python -m venv myenv

# Activate the virtual environment
    Mac/Linux
    source myenv/bin/activate
    Windows
    myenv\Scripts\activate

# Install required packages
   pip install -r requirements.txt

   Upgrade pip (optional but recommended)
   python -m pip install --upgrade pip

# Run the application
   Windows
   python app.py
   Mac/Linux
   python3 app.py
